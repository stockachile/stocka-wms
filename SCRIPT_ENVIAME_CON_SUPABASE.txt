/**
 * FUNCIÓN NUEVA: Exporta los datos que no han sido marcados en la columna R.
 * Se debe programar con un activador cada 4 horas.
 */
function exportarDatosCada4Horas() {
  var ssOrigen = SpreadsheetApp.getActiveSpreadsheet();
  var hojaOrigen = ssOrigen.getSheetByName(NOMBRE_HOJA_DATOS);
  
  // Abrir la planilla de destino por su ID
  var ssDestino = SpreadsheetApp.openById(ID_PLANILLA_DESTINO);
  var hojaDestino = ssDestino.getSheetByName(NOMBRE_HOJA_DESTINO);
  
  var ultimaFila = hojaOrigen.getLastRow();
  if (ultimaFila < 2) return; // No hay datos
  
  // Obtenemos todos los datos (incluyendo la columna R)
  var numColumnas = hojaOrigen.getLastColumn();
  var rangoCompleto = hojaOrigen.getRange(2, 1, ultimaFila - 1, Math.max(numColumnas, COLUMNA_CONTROL));
  var datos = rangoCompleto.getValues();
  
  var filasParaExportar = [];
  var indicesParaMarcar = [];

  for (var i = 0; i < datos.length; i++) {
    var filaActual = datos[i];
    var estadoExportacion = filaActual[COLUMNA_CONTROL - 1]; // Columna R
    
    // Si la columna R está vacía, preparamos para exportar
    if (!estadoExportacion || estadoExportacion.toString().trim() === "") {
      filasParaExportar.push(filaActual);
      indicesParaMarcar.push(i + 2); // Guardamos el número de fila real
    }
  }

  if (filasParaExportar.length > 0) {
    // 1. Enviar datos al destino (se agregan al final)
    hojaDestino.getRange(hojaDestino.getLastRow() + 1, 1, filasParaExportar.length, filasParaExportar[0].length)
               .setValues(filasParaExportar);
    
    // 2. Marcar en el origen como "Exportado" con fecha y hora
    var fechaSello = Utilities.formatDate(new Date(), "GMT-3", "dd/MM/yyyy HH:mm");
    indicesParaMarcar.forEach(function(numFila) {
      hojaOrigen.getRange(numFila, COLUMNA_CONTROL).setValue("Exportado: " + fechaSello);
    });
    
    Logger.log("Exportación exitosa: " + filasParaExportar.length + " filas enviadas.");
  } else {
    Logger.log("Sin datos nuevos para exportar.");
  }
}

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var d = data.delivery || data;
    
    var idEnviameMaestro = (d.identifier || d.id || "").toString().trim();
    if (!idEnviameMaestro) return;

    if (!d.customer || !d.customer.full_name || (!d.company && !d.seller_id)) {
      var dAPI = consultarEnviameAPI(idEnviameMaestro);
      if (dAPI) d = dAPI; 
    }

    procesarEntrada(ss, d, idEnviameMaestro);

    return ContentService.createTextOutput(JSON.stringify({"result": "success"})).setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({"result": "error", "message": error.message})).setMimeType(ContentService.MimeType.JSON);
  }
}

function procesarEntrada(ss, d, idMaestro) {
  var sheet = ss.getSheetByName(NOMBRE_HOJA_DATOS);
  var columnKey = 17; // Columna Q
  var lastRow = sheet.getLastRow();
  var values = lastRow > 1 ? sheet.getRange(1, columnKey, lastRow).getValues() : [];
  var filaEncontrada = -1;
  var fechaAhora = new Date();

  for (var i = 0; i < values.length; i++) {
    if (values[i][0].toString().trim() === idMaestro) {
      filaEncontrada = i + 1;
      break;
    }
  }

  var sellerId = "";
  if (d.company && d.company.id) sellerId = d.company.id.toString();
  else if (d.seller_id) sellerId = d.seller_id.toString();
  else if (d.links && d.links.length > 0) {
    for (var k=0; k < d.links.length; k++) {
      var m = d.links[k].href.match(/companies\/(\d+)/);
      if (m) { sellerId = m[1]; break; }
    }
  }

  var trackingReal = (d.tracking_number || d.carrier_tracking_number || idMaestro).toString().trim();
  var tUrl = (sellerId && trackingReal) ? "https://api.enviame.io/s2/companies/" + sellerId + "/deliveries/" + trackingReal + "/tracking" : "N/A";

  var vendedor = buscarNombreVendedorEnDB(ss, sellerId);
  var idPedido = d.imported_id || d.order_number || "N/A";
  var status = d.status ? (d.status.name || d.status) : "N/A";
  var cliente = d.customer ? d.customer.full_name : "N/A";
  var courier = d.carrier || (d.courier ? d.courier.name : "N/A");
  var ciudad = d.shipping_address ? (d.shipping_address.place || d.shipping_address.city) : "N/A";
  var direccion = d.shipping_address ? d.shipping_address.full_address : "N/A";
  var comuna = d.shipping_address ? (d.shipping_address.county || d.shipping_address.commune) : "N/A";
  var lUrl = d.label ? (d.label.PDF || d.label) : "N/A";

  var tipoServicio = d.shipping_service || (d.carrier ? (d.carrier.service || d.carrier.service_name || d.carrier.name) : null) || d.service || "N/A";
  var clienteNombre = d.customer ? (d.customer.full_name || d.customer.name || "N/A") : "N/A";
  var clienteTelefono = d.customer ? (d.customer.phone || d.customer.telephone || "N/A") : "N/A";
  var clienteEmail = d.customer ? (d.customer.email || "N/A") : "N/A";
  var direccionDestino = d.shipping_address ? (d.shipping_address.full_address || d.shipping_address.street || "N/A") : "N/A";
  var complementoDestino = d.shipping_address ? (d.shipping_address.complement || d.shipping_address.additional_info || "N/A") : "N/A";

  // 1. Guardar en Google Sheets (flujo existente)
  if (filaEncontrada != -1) {
    sheet.getRange(filaEncontrada, 2).setValue(idPedido);
    sheet.getRange(filaEncontrada, 3).setValue(trackingReal);
    sheet.getRange(filaEncontrada, 4).setValue(status);
    sheet.getRange(filaEncontrada, 5).setValue(vendedor);
    sheet.getRange(filaEncontrada, 6).setValue(cliente);
    sheet.getRange(filaEncontrada, 7).setValue(ciudad);
    sheet.getRange(filaEncontrada, 9).setValue(courier);
    sheet.getRange(filaEncontrada, 11).setValue(direccion);
    sheet.getRange(filaEncontrada, 12).setValue(comuna);
    sheet.getRange(filaEncontrada, 14).setValue(tUrl);
    sheet.getRange(filaEncontrada, 15).setValue(lUrl);
    sheet.getRange(filaEncontrada, 16).setValue(fechaAhora);
  } else {
    sheet.appendRow([
      fechaAhora, idPedido, trackingReal, status, vendedor, cliente, ciudad, 1,
      courier, "N/A", direccion, comuna, d.created_at || "N/A", tUrl, lUrl, fechaAhora, idMaestro
    ]);
  }

  // 2. ACTUALIZACIÓN EN TIEMPO REAL EN SUPABASE (NUEVO FLUJO)
  // Se envía incondicionalmente para registrar toda la data que nos llega de Enviame
  var extraData = {
    "seller_name": vendedor !== "N/A" ? vendedor : null,
    "service_type": tipoServicio !== "N/A" ? tipoServicio : null,
    "recipient_name": clienteNombre !== "N/A" ? clienteNombre : null,
    "recipient_phone": clienteTelefono !== "N/A" ? clienteTelefono : null,
    "recipient_email": clienteEmail !== "N/A" ? clienteEmail : null,
    "recipient_address": direccionDestino !== "N/A" ? direccionDestino : null,
    "address_complement": complementoDestino !== "N/A" ? complementoDestino : null,
    "commune": comuna !== "N/A" ? comuna : null
  };
  var orderIdDb = (idPedido && idPedido !== "N/A") ? idPedido : null;
  actualizarPedidoEnSupabase(orderIdDb, trackingReal, tUrl, lUrl, status, courier, idMaestro, d, extraData);
}

function buscarNombreVendedorEnDB(ss, sellerId) {
  if (!sellerId) return "N/A";
  var dbSheet = ss.getSheetByName("Vendedores");
  if (!dbSheet) return "ID: " + sellerId;
  var dbData = dbSheet.getDataRange().getValues();
  for (var i = 1; i < dbData.length; i++) {
    if (dbData[i][0].toString().trim() === sellerId.trim()) return dbData[i][1];
  }
  return "ID: " + sellerId;
}

function consultarEnviameAPI(deliveryId) {
  var url = "https://api.enviame.io/api/s2/v2/deliveries/" + deliveryId;
  var options = { "method" : "get", "headers" : { "Accept": "application/json", "api-key": API_KEY }, "muteHttpExceptions": true };
  try {
    var response = UrlFetchApp.fetch(url, options);
    if (response.getResponseCode() === 200) {
      var json = JSON.parse(response.getContentText());
      return json.data; 
    }
  } catch (e) { console.log("Error API: " + e); }
  return null;
}

/**
 * Mapea los estados de Enviame a los estados internos definidos en WMS STOCKA:
 * 'para procesar', 'en preparación', 'preparado', 'despachado', 'en tránsito', 
 * 'listo para retiro', 'retirado', 'entregado', 'en espera', 'cancelado'
 */
function mapEnviameStatusToWms(enviameStatus) {
  if (!enviameStatus) return null;
  
  var statusLower = enviameStatus.toLowerCase().trim();
  
  // 1. Entregado
  if (statusLower.indexOf("entregado") !== -1 || statusLower === "delivered" || statusLower === "entregado con exito") {
    return "entregado";
  }
  
  // 2. En tránsito / En reparto
  if (statusLower.indexOf("transito") !== -1 || 
      statusLower.indexOf("ruta") !== -1 || 
      statusLower.indexOf("camino") !== -1 || 
      statusLower.indexOf("distribucion") !== -1 || 
      statusLower === "in_transit" ||
      statusLower === "en reparto") {
    return "en tránsito";
  }
  
  // 3. Despachado / Recolectado por Courier
  if (statusLower.indexOf("recolectado") !== -1 || 
      statusLower.indexOf("recogido") !== -1 || 
      statusLower.indexOf("admitido") !== -1 || 
      statusLower.indexOf("despachado") !== -1 || 
      statusLower === "picked_up" || 
      statusLower === "shipped") {
    return "despachado";
  }
  
  // 4. Cancelado / Anulado
  if (statusLower.indexOf("cancelado") !== -1 || 
      statusLower.indexOf("anulado") !== -1 || 
      statusLower === "canceled" || 
      statusLower === "deleted") {
    return "cancelado";
  }
  
  // 5. Incidencia / Novedad (Problemas en entrega)
  if (statusLower.indexOf("novedad") !== -1 || 
      statusLower.indexOf("incidencia") !== -1 || 
      statusLower.indexOf("problema") !== -1 || 
      statusLower.indexOf("fallido") !== -1 ||
      statusLower.indexOf("siniestro") !== -1 ||
      statusLower.indexOf("perdido") !== -1) {
    return "incidencia";
  }

  // 6. En espera / Retenido
  if (statusLower.indexOf("espera") !== -1 || statusLower.indexOf("retenido") !== -1) {
    return "en espera";
  }
  
  return null; // Si no coincide exactamente, no forzamos cambio de estado interno para evitar errores
}

/**
 * Envía una solicitud PATCH a la API de Supabase para actualizar los datos del pedido en tiempo real.
 */
function actualizarPedidoEnSupabase(idPedido, trackingNumber, trackingUrl, labelUrl, status, courier, enviameId, rawDeliveryData, extraData) {
  if (SUPABASE_SERVICE_ROLE_KEY === "INGRESA_AQUI_TU_SUPABASE_SERVICE_ROLE_KEY" || SUPABASE_SERVICE_ROLE_KEY.indexOf("INGRESA_AQUI") === 0) {
    Logger.log("Supabase desactivado: debes configurar la variable SUPABASE_SERVICE_ROLE_KEY.");
    return;
  }

  // Realizar el UPSERT directo en la tabla 'enviame_shipments'
  // Almacenamos el idPedido de texto (ej: Sag_16101) directamente en la columna order_id sin validaciones cruzadas.
  var payload = {
    "id": enviameId || null,
    "order_id": idPedido || null,
    "tracking_number": trackingNumber || null,
    "tracking_url": trackingUrl || null,
    "label_url": labelUrl || null,
    "courier": courier || null,
    "status": status || null,
    "seller_name": (extraData && extraData.seller_name) ? extraData.seller_name : null,
    "service_type": (extraData && extraData.service_type) ? extraData.service_type : null,
    "recipient_name": (extraData && extraData.recipient_name) ? extraData.recipient_name : null,
    "recipient_phone": (extraData && extraData.recipient_phone) ? extraData.recipient_phone : null,
    "recipient_email": (extraData && extraData.recipient_email) ? extraData.recipient_email : null,
    "recipient_address": (extraData && extraData.recipient_address) ? extraData.recipient_address : null,
    "address_complement": (extraData && extraData.address_complement) ? extraData.address_complement : null,
    "commune": (extraData && extraData.commune) ? extraData.commune : null,
    "raw_payload": rawDeliveryData || null,
    "updated_at": new Date().toISOString()
  };

  var upsertUrl = SUPABASE_URL + "/rest/v1/enviame_shipments";
  var upsertOptions = {
    "method": "POST",
    "headers": {
      "apikey": SUPABASE_SERVICE_ROLE_KEY,
      "Authorization": "Bearer " + SUPABASE_SERVICE_ROLE_KEY,
      "Content-Type": "application/json",
      "Prefer": "resolution=merge-duplicates" // Esto fuerza el upsert si la clave primaria 'id' ya existe
    },
    "payload": JSON.stringify(payload),
    "muteHttpExceptions": true
  };

  try {
    var response = UrlFetchApp.fetch(upsertUrl, upsertOptions);
    var responseCode = response.getResponseCode();
    var responseText = response.getContentText();
    Logger.log("Supabase upsert response: [" + responseCode + "] " + responseText);
  } catch (error) {
    Logger.log("Excepción al intentar hacer upsert en Supabase: " + error.toString());
  }
}

/**
 * FUNCIÓN NUEVA: Sincroniza todos los datos históricos existentes en la hoja de Google Sheets
 * directamente a la tabla enviame_shipments en Supabase.
 * Úsala para cargar datos históricos de los últimos 30 días o más que ya están en la hoja.
 */
function sincronizarHojaASupabase() {
  if (SUPABASE_SERVICE_ROLE_KEY === "INGRESA_AQUI_TU_SUPABASE_SERVICE_ROLE_KEY" || SUPABASE_SERVICE_ROLE_KEY.indexOf("INGRESA_AQUI") === 0) {
    Logger.log("Supabase desactivado: debes configurar la variable SUPABASE_SERVICE_ROLE_KEY.");
    return;
  }

  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(NOMBRE_HOJA_DATOS);
  var lastRow = sheet.getLastRow();
  if (lastRow < 2) {
    Logger.log("No hay datos para sincronizar.");
    return;
  }

  // Obtenemos todos los datos (rango de columnas de A a Q)
  var range = sheet.getRange(2, 1, lastRow - 1, 17);
  var data = range.getValues();
  
  var payloads = [];
  var batchSize = 100; // Enviamos en lotes de 100 para no sobrecargar la petición

  for (var i = 0; i < data.length; i++) {
    var row = data[i];
    
    // Columnas:
    // B (index 1) -> idPedido
    // C (index 2) -> trackingReal
    // D (index 3) -> status
    // I (index 8) -> courier
    // N (index 13) -> tUrl
    // O (index 14) -> lUrl
    // Q (index 16) -> idMaestro (PK)
    
    var idMaestro = row[16] ? row[16].toString().trim() : "";
    if (!idMaestro) {
      // Si no hay idMaestro, usamos el trackingReal o generamos un hash temporal
      var trackingReal = row[2] ? row[2].toString().trim() : "";
      if (trackingReal) {
        idMaestro = trackingReal;
      } else {
        continue; // Fila sin identificador válido, omitir
      }
    }

    var idPedido = row[1] ? row[1].toString().trim() : "";
    var trackingReal = row[2] ? row[2].toString().trim() : "";
    var status = row[3] ? row[3].toString().trim() : "";
    var vendedor = row[4] ? row[4].toString().trim() : "";
    var cliente = row[5] ? row[5].toString().trim() : "";
    var courier = row[8] ? row[8].toString().trim() : "";
    var direccion = row[10] ? row[10].toString().trim() : "";
    var comuna = row[11] ? row[11].toString().trim() : "";
    var tUrl = row[13] ? row[13].toString().trim() : "";
    var lUrl = row[14] ? row[14].toString().trim() : "";

    payloads.push({
      "id": idMaestro,
      "order_id": (idPedido && idPedido !== "N/A") ? idPedido : null,
      "tracking_number": (trackingReal && trackingReal !== "N/A") ? trackingReal : null,
      "tracking_url": (tUrl && tUrl !== "N/A") ? tUrl : null,
      "label_url": (lUrl && lUrl !== "N/A") ? lUrl : null,
      "courier": (courier && courier !== "N/A") ? courier : null,
      "status": (status && status !== "N/A") ? status : null,
      "seller_name": (vendedor && vendedor !== "N/A") ? vendedor : null,
      "service_type": null, // El tipo de servicio no se encuentra en las columnas del sheet
      "recipient_name": (cliente && cliente !== "N/A") ? cliente : null,
      "recipient_phone": null, // El teléfono no se encuentra en las columnas del sheet
      "recipient_email": null, // El email no se encuentra en las columnas del sheet
      "recipient_address": (direccion && direccion !== "N/A") ? direccion : null,
      "address_complement": null, // El complemento no se encuentra en las columnas del sheet
      "commune": (comuna && comuna !== "N/A") ? comuna : null,
      "raw_payload": null, // En sincronización masiva de Sheets no tenemos el payload original JSON completo
      "updated_at": new Date().toISOString()
    });

    // Enviar lote si alcanza el batchSize
    if (payloads.length >= batchSize) {
      enviarLoteASupabase(payloads);
      payloads = [];
    }
  }

  // Enviar remanente
  if (payloads.length > 0) {
    enviarLoteASupabase(payloads);
  }

  Logger.log("Sincronización masiva de Sheets a Supabase completada.");
}

/**
 * Función interna para enviar un lote (array) de registros a la tabla rest de Supabase.
 */
function enviarLoteASupabase(batchPayloads) {
  var upsertUrl = SUPABASE_URL + "/rest/v1/enviame_shipments";
  var upsertOptions = {
    "method": "POST",
    "headers": {
      "apikey": SUPABASE_SERVICE_ROLE_KEY,
      "Authorization": "Bearer " + SUPABASE_SERVICE_ROLE_KEY,
      "Content-Type": "application/json",
      "Prefer": "resolution=merge-duplicates, return=minimal"
    },
    "payload": JSON.stringify(batchPayloads),
    "muteHttpExceptions": true
  };

  try {
    var response = UrlFetchApp.fetch(upsertUrl, upsertOptions);
    var responseCode = response.getResponseCode();
    if (responseCode >= 200 && responseCode < 300) {
      Logger.log("Lote de " + batchPayloads.length + " envíos guardado en Supabase con éxito.");
    } else {
      Logger.log("Error al enviar lote: [" + responseCode + "] " + response.getContentText());
    }
  } catch (error) {
    Logger.log("Excepción al intentar hacer upsert por lotes en Supabase: " + error.toString());
  }
}
